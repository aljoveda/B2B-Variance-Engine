Executive Summary

CorDuro didn’t have a way to compare underwriting (Proformas) against actual acquired units (Info Sheets) across its portfolio, resulting in the acquisition of lower-value units that generated less revenue in the long term.

To solve it, a 4-stage automated R pipeline was generated to parse raw Excel files, standardize nomenclature, facilitate matching, calculate statistical variance, and generate PDF deliverables.

As a result, CorDuro saved tens of thousands in Net-Operating Income by catching the deviations early and requesting different units to its partners.

# Normalization Function: Converts "152 Temple St, CT" -> "152 temple"
normalize_address <- function(raw_address) {
  if (is.na(raw_address) || trimws(raw_address) == "") return(NA_character_)
  
  raw_address %>%
    str_to_lower() %>%
    str_replace_all("[^a-z0-9 ]", " ") %>% # Replace punctuation with spaces
    str_squish() %>%                       # Remove extra whitespace
    word(1, 2)                             # Keep ONLY the first two words (e.g., "152 temple")
}

extract_info_sheet <- function(file_path) {
  file_name <- basename(file_path)
  raw_address <- NA_character_
  
  # Attempt 1: Look in 'Additional Info' tab
  add_info <- tryCatch(read_excel(file_path, sheet = "Additional Info", col_names = FALSE), error = function(e) NULL)
  
  # FIX: Added `ncol(add_info) >= 2` to prevent the out-of-bounds error on blank/broken tabs
  if (!is.null(add_info) && nrow(add_info) > 0 && ncol(add_info) >= 2) {
    addr_row <- add_info %>% filter(str_detect(str_to_lower(as.character(...1)), "property address"))
    if (nrow(addr_row) > 0 && !is.na(addr_row[[2]][1])) {
      raw_address <- as.character(addr_row[[2]][1])
    }
  }
  
  # Attempt 2: If Attempt 1 failed (e.g. Expansions), check 'Home Information'
  if (is.na(raw_address) || trimws(raw_address) == "") {
    home_info <- tryCatch(read_excel(file_path, sheet = "Home Information", n_max = 10, col_names = FALSE), error = function(e) NULL)
    if (!is.null(home_info)) {
      # Find which column contains "Unit Mailing Address" in the first few rows
      col_idx <- which(str_detect(str_to_lower(as.character(home_info[2, ])), "mailing address") | 
                         str_detect(str_to_lower(as.character(home_info[3, ])), "mailing address"))[1]
      
      if (!is.na(col_idx)) {
        # Grab the first non-NA address listed below the header
        address_col <- as.character(home_info[4:10, col_idx]) 
        raw_address <- address_col[!is.na(address_col)][1]
      }
    }
  }
  
  tibble(
    Info_Sheet_Name = file_name,
    Address_Key = normalize_address(raw_address)
  )
}

# The Peek Function
peek_proforma_name <- function(file_path, first_valid_tab) {
  raw_data <- suppressMessages(read_excel(file_path, sheet = first_valid_tab, n_max = 20, col_names = FALSE))
  for (col_idx in 1:min(3, ncol(raw_data))) {
    name_row_idx <- which(str_detect(as.character(raw_data[[col_idx]]), "Financial Summary"))[1]
    if (!is.na(name_row_idx)) {
      raw_name <- as.character(raw_data[name_row_idx, col_idx])
      return(str_remove(raw_name, "Landing Autopilot Financial Summary: |Landing Financial Summary: ") %>% trimws())
    }
  }
  return(NA_character_)
}

extract_proforma_detailed <- function(file_path) {
  all_tabs <- excel_sheets(file_path)
  valid_tabs <- all_tabs[!str_detect(str_to_lower(all_tabs), "market overview")]
  
  if (length(valid_tabs) == 0) return(NULL)
  
  target_tab <- valid_tabs[1] # Default
  
  # If multiple tabs exist, apply intelligent routing
  if (length(valid_tabs) > 1) {
    prof_name <- peek_proforma_name(file_path, valid_tabs[1])
    term <- NA
    
    if (!is.na(prof_name)) {
      term <- prof_to_term %>% filter(Prof_Name == prof_name) %>% pull(Term) %>% .[1]
    }
    
    if (!is.na(term)) {
      target_str <- case_when(
        term %in% c(3, 12) ~ "1 year",
        term == 15 ~ "lease",
        term == 24 ~ "2 year",
        term == 36 ~ "3 year",
        term == 48 ~ "4 year",
        term == 60 ~ "5 year",
        TRUE ~ "1 year"
      )
      
      matched_tab <- valid_tabs[str_detect(str_to_lower(valid_tabs), target_str)]
      if (length(matched_tab) > 0) target_tab <- matched_tab[1]
    }
  }
  
  # Read the target tab safely
  raw_proforma <- tryCatch(suppressMessages(read_excel(file_path, sheet = target_tab, col_names = FALSE)), error = function(e) NULL)
  if (is.null(raw_proforma)) return(NULL)
  
  # DYNAMIC FIX: Find Deal Name across the first 3 columns
  deal_name <- NA_character_
  scraped_address <- NA_character_
  col_idx_found <- 1
  
  for (col_idx in 1:min(3, ncol(raw_proforma))) {
    name_row_idx <- which(str_detect(as.character(raw_proforma[[col_idx]]), "Financial Summary"))[1]
    if (!is.na(name_row_idx)) {
      deal_name <- str_remove(as.character(raw_proforma[name_row_idx, col_idx]), "Landing Autopilot Financial Summary: |Landing Financial Summary: ") %>% trimws()
      row_address_data <- as.character(raw_proforma[name_row_idx + 1, ])
      raw_address <- row_address_data[!is.na(row_address_data) & trimws(row_address_data) != ""][1]
      scraped_address <- str_remove_all(raw_address, "[\r\n]")
      col_idx_found <- col_idx
      break
    }
  }
  
  if (is.na(deal_name)) deal_name <- paste("ERROR/BLANK -", basename(file_path))
  if (ncol(raw_proforma) < (col_idx_found + 2)) return(NULL) # Safety check
  
  # Extract Data maintaining column relativity
  raw_proforma %>%
    select(Unit_Type = all_of(col_idx_found), Proforma_Units = all_of(col_idx_found + 1), Proforma_Rent = all_of(col_idx_found + 2)) %>%
    filter(as.character(Unit_Type) %in% c("Studio", "1 Bedroom", "2 Bedroom", "3 Bedroom", "4 Bedroom", "Total")) %>%
    mutate(
      Deal_Name = deal_name,
      Property_Address = replace_na(scraped_address, "Address Not Found"), 
      Proforma_Units = suppressWarnings(replace_na(as.numeric(Proforma_Units), 0)),
      Proforma_Rent = suppressWarnings(replace_na(as.numeric(Proforma_Rent), 0)),
      Source_File = basename(file_path),
      Target_Tab = target_tab
    ) %>%

salespeople_with_expansions <- statistical_db %>%

if (length(salespeople_with_expansions) > 0) {
  expansion_stats_df <- statistical_db %>%
    filter(Salesperson %in% salespeople_with_expansions) %>%
    mutate(Deal_Type = if_else(str_detect(Deal_Name, " - Expansion"), "Expansion", "New Deal")) %>%
    group_by(Salesperson, Deal_Type) %>%
    summarise(
      Deals = n(), 
      raw_mean = mean(Rent_Variance_Pct, na.rm = TRUE), 
      raw_sd = sd(Rent_Variance_Pct, na.rm = TRUE),
      `Worst Miss` = percent(min(Rent_Variance_Pct, na.rm = TRUE), 0.1), 
      `Best Beat` = percent(max(Rent_Variance_Pct, na.rm = TRUE), 0.1),
      t_margin = if_else(Deals > 1, qt(0.975, df = pmax(1, Deals - 1)) * (raw_sd / sqrt(Deals)), NA_real_),
      .groups = "drop"
    ) %>%
    mutate(
      Mean = percent(raw_mean, 0.1), 
      `Std Dev` = if_else(is.na(raw_sd), "N/A", percent(raw_sd, 0.1)),
      `95% CI Low` = if_else(is.na(t_margin), "N/A", percent(raw_mean - t_margin, 0.1)), 
      `95% CI High` = if_else(is.na(t_margin), "N/A", percent(raw_mean + t_margin, 0.1))
    ) %>%
    select(Salesperson, Type = Deal_Type, Deals, raw_mean, Mean, `Std Dev`, `95% CI Low`, `95% CI High`, `Worst Miss`, `Best Beat`) %>%
    arrange(Salesperson, Type)
  
  dumbbell_plot <- expansion_stats_df %>%
    ggplot(aes(x = raw_mean, y = reorder(Salesperson, raw_mean, FUN = mean), color = Type)) +
    geom_line(aes(group = Salesperson), color = "gray75", linewidth = 1.5) +
    geom_point(size = 5, alpha = 0.9) +
    geom_vline(xintercept = port_mean, linetype = "dashed", color = "darkred", linewidth = 1) +
    scale_x_continuous(labels = percent_format(accuracy = 1)) +
    scale_color_manual(values = c("Expansion" = "darkorange", "New Deal" = "steelblue")) +
    labs(title = "Rent Variance: Expansions vs. New Deals", subtitle = paste0("Testing the 'Expansion Inflation' Hypothesis | Dashed line = Portfolio Avg (", percent(port_mean, 0.1), ")"), y = NULL, x = "Average Rent Variance Percentage", color = "Deal Type") +
    theme_minimal(base_size = 14) +

  grid.arrange(dumbbell_plot, tableGrob(final_table_df, rows = NULL, theme = exp_theme), heights = c(2.5, 1.2))
